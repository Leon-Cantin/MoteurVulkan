cmake_minimum_required(VERSION 3.15)

set( TARGET_NAME PBR_3D_Renderer )
				  
file( GLOB SRCS RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} sources/* )

add_executable( ${TARGET_NAME} ${SRCS} )

target_compile_definitions( ${TARGET_NAME} PRIVATE -DUNICODE )
target_compile_definitions( ${TARGET_NAME} PRIVATE -D_UNICODE )

target_compile_features( ${TARGET_NAME} PUBLIC cxx_std_20)

target_include_directories( ${TARGET_NAME} PRIVATE sources )

target_link_libraries( ${TARGET_NAME} PUBLIC Renderer )
target_link_libraries( ${TARGET_NAME} PUBLIC WindowHandler )
target_link_libraries( ${TARGET_NAME} PUBLIC Utils )
